导航栏下划线鼠标跟随效果插件

—— 原生Javascript，不依赖任何三方库
使用 nav-underline.js
            
                <!-- 引入 nav-underline.js -->
                <script src="src/nav-underline.js"></script>

                <ul class="nav">
                    <li><a href="javascript:;">首页</a></li>
                    <li><a href="javascript:;">产品中心</a></li>
                    <li><a href="javascript:;">关于我们</a></li>
                    <li><a href="javascript:;">联系我们</a></li>
                </ul>
  • 首页
  • 产品中心
  • 关于我们
  • 联系我们
            
                NavUnderline('.nav');
                // 或
                new NavUnderline('.nav');
  • 首页
  • 产品中心
  • 关于我们
  • 联系我们
            
                NavUnderline('#nav2', {
                    duration: '1s',
                    timing: 'ease',
                    textColor: '#8FBC94',
                    lineColor: '#C5E99B'
                });
                // 或
                new NavUnderline('#nav2', {
                    duration: '1s',         // 过渡时间
                    timing: 'ease',         // 速度曲线 linear,ease,ease-in,ease-out,ease-in-out
                    textColor: '#8FBC94',   // 导航栏文字颜色
                    lineColor: '#C5E99B'    // 下划线颜色
                });
